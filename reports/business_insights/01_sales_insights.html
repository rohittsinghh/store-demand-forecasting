store_perf = df.groupby("store")["sales"].mean().sort_values(ascending=False)

store_perf.plot(
    kind="bar",
    figsize=(10,4),
    title="Average Sales per Store"
)
plt.ylabel("Average Sales")
plt.show()

df["day_type"] = df.date.dt.dayofweek.apply(
    lambda x: "Weekend" if x >= 5 else "Weekday"
)

df.groupby("day_type")["sales"].mean().plot(
    kind="bar",
    title="Weekend vs Weekday Sales"
)
plt.ylabel("Average Sales")
plt.show()
